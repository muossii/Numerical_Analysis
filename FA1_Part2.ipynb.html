68   def bisecting_iter_num1(iter, an, bn):
69       
70       # create dictionary to store iteration data
71       val_table_dict = {'n': [], 'an': [], 'bn': [], 'pn': [], 
72                           'f(pn)': []}
73       
74       # starting loop
75       i = 0
76       
77       while i <= iter:
78           
79           
80           # setting n and pn for each iteration
81           n = i+1
82           pn = (an+bn)/2
83           
84           fpn = (np.exp(pn)-2)-np.cos(np.exp(pn)-2)
85           
86           val_table_dict['n'].append(n)
87           val_table_dict['an'].append(an)
88           val_table_dict['bn'].append(bn)
89           val_table_dict['pn'].append(pn)
90           val_table_dict['f(pn)'].append(fpn)
91           
92           # break the loop if solution found earlier than expected:
93           
94           if fpn == 0:
95               break
96               
97           # compare signs
98           
99           fan = (np.exp(an)-2)-np.cos(np.exp(an)-2)
100          fbn = (np.exp(bn)-2)-np.cos(np.exp(bn)-2)
101          
102          if (fan>0 and fpn>0) or (fan<0 and fpn<0):
103              an = pn
104          elif (fbn>0 and fpn>0) or (fbn<0 and fpn<0):
105              bn = pn
106          
107          i=i+1
108      
109      return val_table_dict

135  def fixed_point_iter(iter, pn ):
136      
137      # create dictionary to store iteration data
138      val_table_dict = {'n': [], 'pn': []}
139      
140      # starting loop
141      i = 0
142      
143      while i <= iter:
144          
145          # get n, and pn+1
146          n = i+1
147          p = (5/(pn**2))+2
148          
149          # append 
150          val_table_dict['n'].append(n)
151          val_table_dict['pn'].append(p)
152          
153          if (abs(p-pn))==0:
154              break
155          
156          pn = p
157          
158          i=i+1
159          
160      return val_table_dict
